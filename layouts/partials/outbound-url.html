Input: dict "url" "https://example.com" "campaign" "product_page" (optional) Appends ?utm_source=ailist&utm_medium=referral&utm_campaign= */}} {{- $url := .url -}} {{- $campaign := .campaign | default "directory" -}} {{- $separator := "?" -}} {{- if strings.Contains $url "?" -}} {{- $separator = "&" -}} {{- end -}} {{- $result := printf "%s%sutm_source=ailist&utm_medium=referral&utm_campaign=%s" $url $separator $campaign | safeURL -}} {{- return $result -}}
